$$|\langle X, Y\rangle|\leq|\langle X,X\rangle\langle Y,Y\rangle|^{\frac 12}$$

\subsubsection*{Random Telegraph Signal}

Random flip.

Discrete Possion Distribution:
$$P(X=k)=\frac{(\lambda t)^k}{k!}\exp(-\lambda t)$$

Between moments s and t, there is $k$ flip times $P([s, t]=k)$.

$$\begin{aligned}
  R_X(t,s)&=E(X(t)X(s)) \\
          &=P(X(t)X(s)>0) - P(X(t)X(s)<0)\\
          &=P([s,t]\text{ is even}) - P([s,t]\text{ is odd})\\
          &=\sum_{\text{k is even}}P([s,t]=k) -\sum_{\text{k is odd}}P([s,t]=k) \\
          &=\sum_{\text{k is even}}\frac{\lambda(t-s)^k}{k!}\exp(-\lambda(t-s)) - \sum_{\text{k is odd}}\frac{\lambda(t-s)^k}{k!}\exp(-\lambda(t-s))\\
          &=\frac 12\exp(-\lambda(t-s))\Bigl(\bigl(\exp(\lambda(t-s))+\exp(-\lambda(t-s))\bigr)-\bigl(\exp(\lambda(t-s))-\exp(-\lambda(t-s))\bigr)\Bigr)\\
          &=\exp(-\lambda(t-s))\Rightarrow \exp(-\lambda|t-s|)
\end{aligned}$$

\subsubsection*{Correlation Function is Positive Definite}
$f(t)$ is P.d. $\Longleftrightarrow \forall n, \forall t_1, \cdots, t_n, (f(t_i-t_j))_{ij}\geq 0$

Now we prove the Correlated function is P.d..

$(R_X(t_i-t_j))_{ij}=(E(X(t_i)X(t_j)))_{ij}$

If $X=(X(t_1), \cdots, X(t_n))^T$, then $E(XX^T)=R$ is a positive matrix.

Definition of Positive Definite: $\forall \alpha\in \mathbb{R}^n, \alpha^T R\alpha=\alpha^TE(XX^T)\alpha=E(\alpha^T XX^T\alpha)\geq 0$

$$\begin{aligned}
  &n = 1, t,  R(t-t)=R(0)\geq 0\\
  &n=2, 0, \tau,
  \begin{bmatrix}
    R(0) & R(\tau) \\
    R(-\tau) & R(0)
  \end{bmatrix} \geq 0 \Rightarrow R^2(0)-R^2(t)\geq 0 \Rightarrow |R(t)| \leq R(0)
\end{aligned}$$

e.g.
$R(0)-R(\tau)\geq \frac{1}{4^n}(R(0)-R(2^n\tau))$


$\begin{aligned}
  n=1, R(0)-R(\tau)&\geq \frac{1}{4}(R(0)-R(2\tau)) \\
                   R(2\tau)-4R(\tau)+3R(0)&\geq 0
\end{aligned}$

$
\textbf{Construct}:\qquad\begin{bmatrix}
  R(0) & R(\tau) & R(2\tau) \\
  R(-\tau) & R(0) & R(\tau) \\
  R(-2\tau) & R(-\tau) & R(0)
\end{bmatrix} \geq 0
$

$
\textbf{Known}:\qquad
\begin{aligned}
\begin{bmatrix}x&y&z\end{bmatrix}
 \begin{bmatrix}
  R(0) & R(\tau) & R(2\tau) \\
  R(-\tau) & R(0) & R(\tau) \\
  R(-2\tau) & R(-\tau) & R(0)
  \end{bmatrix} \begin{bmatrix}x\\y\\z\end{bmatrix} &\geq 0 \\
  g_1(x,y,z)R(0) + g_2(x,y,z)R(\tau)+g_3(x,y,z)R(2\tau)&\geq 0
\end{aligned}
$

It's difficult to solve if the function is not linear.

We only have properties of positive definite function. It's not easy to recognize a function that is not definite.

      \frac{1}{2\pi}\int_{-\infty}^{+\infty}S_X(\omega)\bigl(3\cos(0) - 4\cos(\omega \tau) + \cos(2\omega\tau)\bigr)d\omega &\geq 0 \\
\frac{1}{2\pi}\int_{-\infty}^{+\infty}S_X(\omega)\bigl(3 - 4\cos(\omega \tau) + 2\cos^2(\omega\tau)-1\bigr)d\omega &\geq 0 \\
\frac{1}{2\pi}\int_{-\infty}^{+\infty}S_X(\omega)\bigl(2 - 4\cos(\omega \tau) + 2\cos^2(\omega\tau)\bigr)d\omega &\geq 0 \\
\frac{1}{2\pi}\int_{-\infty}^{+\infty}S_X(\omega)2(\cos(\omega\tau)-1)^2d\omega &\geq 0
  \end{aligned}$$
\end{enumerate}

Suppose $X(t)$ is wide-sense stationary:

\begin{enumerate}
  \item Is $Y(t)$ wide-sense stationary?
  \item What's the relation of the Power Spectrum Density between $X(t)$ and $Y(t)$?
\end{enumerate}

$\begin{aligned}
  R_Y(t,s)&=E(Y(t)Y(s)) \\
          &=E(\int_{\infty}^{+\infty}h(t-\tau)X(\tau)d\tau)(\int_{-\infty}^{+\infty}h(s-r)X(r)dr) \\
          &=\int_{\infty}^{+\infty}\int_{\infty}^{+\infty}h(t-\tau)h(s-r)R_X(\tau-r)drds \\
          &=\int_{\infty}^{+\infty}\int_{\infty}^{+\infty}h(t-\tau)\tilde h(r-s)R_X(\tau-r)drds \\
          &=(h\circledast\tilde h\circledast R_X)(t-s)
\end{aligned}$

$$\begin{aligned}
  \int_{-\infty}^{+\infty}\tilde h(t)\exp(-j\omega t) dt
  &=\int_{-\infty}^{+\infty}h(-t)\exp(-j\omega t) dt \\
  &=\int_{-\infty}^{+\infty}h(t)\exp(j\omega t) dt \\
  &=\overline{\int_{-\infty}^{+\infty}h(t)\exp(-j\omega t) dt} \\
  &=\overline{H(\omega)}
\end{aligned}$$

$S_Y(\omega)=H(\omega)\cdot\overline{H(\omega)}S_X(\omega)=|H(\omega)|^2S_X(\omega)$
